\newcommand{\addplotStatistics}[1]{
	   \addplot+[draw=none,fill=dblue2 ]
	   table[ y=total_values_count, col sep=comma, x=col_index] {#1};
     \label{filled}; 

     \addplot+[draw=none, fill=mvcolor] 
	   table[ y=missing_values_count, col sep=comma, x=col_index] {#1};	   
	   \label{missed};       
};

};
